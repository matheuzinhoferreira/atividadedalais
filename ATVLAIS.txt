SELECT SUM(ESTOQUE) AS MEDICAMENTO_ESTOQUE
FROM MEDICAMENTOS m;

SELECT AVG(EXTRACT(YEAR FROM CURRENT_DATE) - EXTRACT(YEAR FROM t.DATA_NASCIMENTO)) AS MEDIA_IDADE
FROM TUTOR t;

SELECT MIN(m.PRECO) AS MENOR_PRECO
FROM MEDICAMENTOS m;

SELECT MAX(p.PESO) AS MAIOR_PESO
FROM PACIENTE p;
	
SELECT COUNT(*) AS TOTAL_TUTORES
FROM TUTOR t;

SELECT m.UNIDADE, SUM(m.ESTOQUE * m.PRECO) AS VALOR_TOTAL_ESTOQUE
FROM MEDICAMENTOS m
GROUP BY m.UNIDADE;

SELECT ir.ID_RECEITA, SUM(ir.TOTAL_ITEM) AS VALOR_TOTAL_RECEITA
FROM ITENSRECEITA ir
GROUP BY ir.ID_RECEITA
HAVING SUM(ir.TOTAL_ITEM) > 100;
