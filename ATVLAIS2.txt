SELECT 
    p.*, 
    t.nome AS nome_tutor, 
    t.endereco AS endereco_tutor, 
    t.telefone AS telefone_tutor
FROM 
    paciente p
LEFT JOIN 
    tutor t ON p.id_tutor = t.id_tutor;
